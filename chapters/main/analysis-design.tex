\chapter{Phân tích và thiết kế hệ thống}

\section{Kiến trúc tổng thể}

Hệ thống ShopNexus bao gồm 5 thành phần chính, giao tiếp với nhau qua
các giao thức TCP và HTTP:

\begin{enumerate}
\item \textbf{Thiết bị nhúng ESP32-CAM:} Thu âm thanh (INMP441) và hình ảnh (OV5640),
  truyền dữ liệu real-time đến server qua TCP.
\item \textbf{Server trung gian (Node.js):} Cầu nối giữa ESP32 và trình duyệt,
  relay TCP↔WebSocket và proxy API đến các dịch vụ AI.
\item \textbf{Dịch vụ khử nhiễu (Noise Reduction):} Mô hình DTLN xử lý khử nhiễu
  giọng nói, triển khai trên Flask.
\item \textbf{Dịch vụ nhận dạng giọng nói (Voice Rephraze):} Chuyển đổi giọng nói
  thành văn bản và sinh mô tả sản phẩm, triển khai trên FastAPI.
\item \textbf{Dịch vụ phân loại sản phẩm (Product Classification):} Phân loại sản phẩm
  vào 32 danh mục sử dụng XLM-RoBERTa, triển khai trên FastAPI.
\end{enumerate}

\section{Luồng xử lý chính}

\begin{enumerate}
\item ESP32 thu âm thanh từ microphone INMP441 (16-bit PCM, 16 kHz, mono) và gửi
  qua TCP port 3002 đến server trung gian.
\item Server trung gian (Node.js) nhận stream PCM, đóng gói thành file WAV và gửi
  đến \textbf{Noise Reduction Service} (port 5000).
\item Audio đã khử nhiễu được gửi đến \textbf{Voice Rephraze Service} (port 5001)
  để chuyển thành text và sinh mô tả sản phẩm theo phong cách viết được chọn.
\item Text mô tả được gửi đến \textbf{Product Classification Service} (port 5002)
  để phân loại vào 1 trong 32 danh mục thương mại điện tử.
\item Kết quả hiển thị trên giao diện web: spectrogram trước/sau khử nhiễu,
  text nhận dạng, mô tả đã sinh, và top-K danh mục dự đoán.
\end{enumerate}

\subsection{Cấu hình chân kết nối}

\textbf{Camera OV5640} sử dụng bus dữ liệu 8-bit song song:
D0--D7 (GPIO 5, 18, 19, 21, 36, 39, 34, 35), XCLK (GPIO 0), PCLK (GPIO 22),
VSYNC (GPIO 25), HREF (GPIO 23), SDA (GPIO 26), SCL (GPIO 27), PWDN (GPIO 32).

\textbf{Microphone INMP441} kết nối qua I2S:
WS -- Word Select (GPIO 2), SCK -- Serial Clock (GPIO 14), SD -- Serial Data (GPIO 15),
sử dụng I2S port 1.

\section{Thiết kế dịch vụ khử nhiễu}

Dịch vụ nhận file WAV chứa giọng nói lẫn nhiễu từ ESP32 và trả về audio đã khử nhiễu.
Triển khai trên Flask 2.3+ với TensorFlow 2.13+, kiến trúc layered (API → Service → Model).

\begin{itemize}
\item \textbf{API:} \texttt{POST /denoise} -- multipart/form-data, file WAV tối đa 50 MB.
\item \textbf{Health check:} \texttt{GET /health} -- trạng thái model và thông số.
\item \textbf{Singleton Pattern:} Model được load một lần khi khởi động, cache trong bộ nhớ.
\end{itemize}

\section{Thiết kế dịch vụ nhận dạng giọng nói}

Dịch vụ chuyển đổi giọng nói thành văn bản và sinh mô tả sản phẩm theo
phong cách viết tùy chọn. Triển khai trên FastAPI với AsyncOpenAI client.

\begin{itemize}
\item \textbf{Speech-to-Text:} Sử dụng OpenAI gpt-4o-transcribe (cloud API),
  hỗ trợ tiếng Việt đa vùng miền.
\item \textbf{Sinh mô tả:} Sử dụng GPT-4o-mini với prompt engineering,
  sinh mô tả 3--5 câu theo phong cách viết được chọn.
\item \textbf{Quản lý phong cách:} Lưu trữ trong SQLite, người dùng có thể
  thêm phong cách mới qua API.
\end{itemize}

\section{Thiết kế server trung gian}

Server trung gian chạy trên Node.js/Express đóng vai trò cầu nối giữa ESP32 và
trình duyệt, với các chức năng chính:

\begin{itemize}
\item \textbf{TCP Camera Server (port 3001):} Nhận JPEG frame từ ESP32,
  parse 4-byte length header, broadcast qua WebSocket \texttt{/camera} đến browser.
\item \textbf{TCP Audio Server (port 3002):} Nhận stream 16-bit PCM từ ESP32,
  duy trì byte-alignment, broadcast qua WebSocket \texttt{/audio} đến browser.
\item \textbf{HTTP API Proxy:} Chuyển tiếp request từ browser đến các AI services:
  \begin{itemize}
  \item \texttt{POST /api/process-voice} → Denoise Service → STT Service
  \item \texttt{POST /api/gen-description} → Voice Rephraze Service
  \item \texttt{POST /api/classify} → Classification Service
  \end{itemize}
\item \textbf{Giao diện web:} Hiển thị video streaming real-time, biểu đồ amplitude
  âm thanh (30 bars), spectrogram trước/sau khử nhiễu (WaveSurfer.js), kết quả
  nhận dạng và phân loại sản phẩm.
\end{itemize}
